🔧 CUSTOMIZATION
════════════════════════════════════════════════════════════════════

Change Primary Color:
  :root {
    --primary: #YOUR_HEX_COLOR;
  }

Modify Button Size:
  .btn-modern {
    padding: 1rem 2rem;
  }

Adjust Grid Columns:
  .isotope-grid {
    grid-template-columns: repeat(auto-fill, minmax(300px, 1fr));
  }

Change Animation Duration:
  :root {
    --transition: all 0.5s ease;
  }
